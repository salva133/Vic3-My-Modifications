on_monthly_pulse_country_mym_random_character_events = {
	random_events = {
		95 = hp_lovecraft.1
		5 = hp_lovecraft.2
	}
}

on_monthly_pulse_country_mym_character_events = {
	events = {
		stoker.1
		stoker.2

		hp_lovecraft.3

		milei.1
	}
}

on_monthly_pulse_country_mym_events = {
	events = {
		anarchist_balkanisation.1
	}
	effect = {
		# Keep modifier_has_unhealthy_economy in sync with the economy flag.
		if = {
			limit = {
				has_healthy_economy = yes
				has_modifier = modifier_has_unhealthy_economy
			}
			remove_modifier = modifier_has_unhealthy_economy
		}
		else_if = {
			limit = {
				has_healthy_economy = no
				NOT = { has_modifier = modifier_has_unhealthy_economy }
			}
			add_modifier = modifier_has_unhealthy_economy
		}
	}
}

on_monthly_pulse_country_mym_heads_or_tails = {
	events = {
		heads_or_tails.1
	}
}

on_monthly_pulse_country = {
	on_actions = {
		delay = { days = 1 }
		on_monthly_pulse_country_mym_random_character_events
		delay = { days = 7 }
		on_monthly_pulse_country_mym_character_events
		delay = { days = 14 }
		on_monthly_pulse_country_mym_events
		delay = { days = 21 }
		on_monthly_pulse_country_mym_heads_or_tails
	}
}

# Hand the global religion tracker journal entry to every country present at game start.
on_game_started_after_lobby = {
	effect = {
		every_country = {
			limit = {
				NOT = { has_journal_entry = je_mym_global_religion_tracker }
			}
			add_journal_entry = {
				type = je_mym_global_religion_tracker
			}
		}
		# Seed the cached heritage stats once so the JE has something to show before
		# the first monthly pulse fires.
		mym_refresh_global_religion_stats = yes
	}
}

# Make sure newly formed / released countries also receive the tracker.
on_country_formed = {
	effect = {
		if = {
			limit = {
				NOT = { has_journal_entry = je_mym_global_religion_tracker }
			}
			add_journal_entry = {
				type = je_mym_global_religion_tracker
			}
		}
	}
}

# Recompute the cached per-heritage global stats once per month, no matter how
# many countries hold the tracker journal entry.
on_monthly_pulse = {
	effect = {
		mym_refresh_global_religion_stats = yes
	}
}
