namespace = baltic_dissolution

# Fires when je_mym_baltic_dissolution completes.
# Creates LIT, LAT, EST from UBD's owned Baltic homeland states,
# sets up bilateral pacts and +200 opinion between all created nations,
# then asks the player which country to continue as.
baltic_dissolution.1 = {
	type = country_event
	placement = root

	title  = baltic_dissolution.1.tt
	desc   = baltic_dissolution.1.dd
	flavor = baltic_dissolution.1.ff

	duration = 1

	event_image = {
		video = "europenorthamerica_springtime_of_nations"
	}

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	immediate = {
		# Republicanise the dissolving Duchy before it splits: a Baltic
		# monarchy, social monarchy or theocracy gives way to a presidential
		# republic. root is still UBD here, so the entity continued via
		# change_tag below carries the new governance law over into LAT.
		if = {
			limit = {
				OR = {
					has_law = law_type:law_monarchy
					has_law = law_type:law_social_monarchy
					has_law = law_type:law_theocracy
				}
			}
			activate_law = law_type:law_presidential_republic
		}
		if = {
			limit = { exists = c:LAT }
			play_as = c:LAT
		}
		else = {
			change_tag = LAT
		}
		# --- Create Lithuania from UBD-owned Lithuanian homeland states ---
		if = {
			limit = {
				NOT = { exists = c:LIT }
				any_scope_state = { state_region = { is_homeland = cu:lithuanian } }
			}
			random_scope_state = {
				limit = { state_region = { is_homeland = cu:lithuanian } }
				save_scope_as = lit_capital_var
			}
			create_country = {
				tag    = LIT
				origin = root
				state  = scope:lit_capital_var
			}
			# Take over the dissolving Duchy's now-republican law set.
			c:LIT ?= { copy_laws = root }
		}
		# Transfer remaining UBD-owned Lithuanian homeland states to LIT.
		if = {
			limit = {
				exists = c:LIT
				any_scope_state = { state_region = { is_homeland = cu:lithuanian } }
			}
			every_scope_state = {
				limit = { state_region = { is_homeland = cu:lithuanian } }
				set_state_owner = c:LIT
			}
		}

		# --- Create Latvia from UBD-owned Latvian homeland states ---
		if = {
			limit = {
				NOT = { exists = c:LAT }
				any_scope_state = { state_region = { is_homeland = cu:latvian } }
			}
			random_scope_state = {
				limit = { state_region = { is_homeland = cu:latvian } }
				save_scope_as = lat_capital_var
			}
			create_country = {
				tag    = LAT
				origin = root
				state  = scope:lat_capital_var
			}
		}
		if = {
			limit = {
				exists = c:LAT
				any_scope_state = { state_region = { is_homeland = cu:latvian } }
			}
			every_scope_state = {
				limit = { state_region = { is_homeland = cu:latvian } }
				set_state_owner = c:LAT
			}
		}

		# --- Create Estonia from UBD-owned Estonian homeland states ---
		if = {
			limit = {
				NOT = { exists = c:EST }
				any_scope_state = { state_region = { is_homeland = cu:estonian } }
			}
			random_scope_state = {
				limit = { state_region = { is_homeland = cu:estonian } }
				save_scope_as = est_capital_var
			}
			create_country = {
				tag    = EST
				origin = root
				state  = scope:est_capital_var
			}
			# Take over the dissolving Duchy's now-republican law set.
			c:EST ?= { copy_laws = root }
		}
		if = {
			limit = {
				exists = c:EST
				any_scope_state = { state_region = { is_homeland = cu:estonian } }
			}
			every_scope_state = {
				limit = { state_region = { is_homeland = cu:estonian } }
				set_state_owner = c:EST
			}
		}

		# Incorporate every state the successor nations hold — states gained via
		# create_country / set_state_owner are not incorporated by default.
		c:LIT ?= { every_scope_state = { set_state_type = incorporated } }
		c:LAT ?= { every_scope_state = { set_state_type = incorporated } }
		c:EST ?= { every_scope_state = { set_state_type = incorporated } }

		# LAT continued from the dissolving Duchy via change_tag and so still
		# carries UBD's primary cultures. Drop every primary culture that isn't
		# part of Latvia's own country definition, then guarantee Latvian is
		# primary. Done here in immediate so it always runs, regardless of which
		# successor the player continues as in the options below.
		c:LAT ?= {
			every_primary_culture = {
				limit = {
					NOT = { prev = { country_definition_has_culture = prev } }
				}
				prev = { remove_primary_culture = prev }
			}
			add_primary_culture = cu:latvian
		}

		# --- Bilateral pacts + maximum opinion between every surviving pair ---
		# The shared article list lives in mym_create_baltic_pact; each if-guard
		# forms a pact only when both partners actually came into existence.
		if = {
			limit = { exists = c:LIT  exists = c:LAT }
			mym_create_baltic_pact = { FIRST = LIT  SECOND = LAT }
		}
		if = {
			limit = { exists = c:LIT  exists = c:EST }
			mym_create_baltic_pact = { FIRST = LIT  SECOND = EST }
		}
		if = {
			limit = { exists = c:LAT  exists = c:EST }
			mym_create_baltic_pact = { FIRST = LAT  SECOND = EST }
		}
	}

	# Player chooses which newly independent nation to continue as.
	option = {
		name = baltic_dissolution.1.a
		trigger = {
			exists = c:LIT
		}
		play_as = c:LIT
	}

	option = {
		name = baltic_dissolution.1.b
		trigger = {
			exists = c:LAT
		}
		play_as = c:LAT
	}

	option = {
		name = baltic_dissolution.1.c
		trigger = {
			exists = c:EST
		}
		play_as = c:EST
	}
}
