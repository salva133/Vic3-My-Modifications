assign_state_name_region_east_anglia = {
    if = {
        limit = {
            state_region = {
                s:STATE_EAST_ANGLIA ?= this
            }
        }
        STATE_EAST_ANGLIA_state_name_assign = yes
    }
}

STATE_EAST_ANGLIA_state_name_assign = {
	if = {
        limit = {
            owner = {
                any_primary_culture = {
                    cu:albionian ?= this
                }
            }
        }

        set_state_name = STATE_EAST_ANGLIA_albionian
        set_hub_names = albionian
    }
	else = { # By putting an else at the end that resets the names we ensure that the names are switched back to their defaults if anyone else takes the states
		reset_state_name = yes
		reset_hub_names = yes
	}
}
