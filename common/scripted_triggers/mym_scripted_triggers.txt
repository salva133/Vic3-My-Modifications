ideology_anti_feminist_valid_trigger = {
	OR = {
		is_interest_group_type = ig_rural_folk
		is_interest_group_type = ig_petty_bourgeoisie
		is_interest_group_type = ig_industrialists
		is_interest_group_type = ig_landowners
	}
}

is_possible_ideology_fascist = {
	has_technology_researched = mass_propaganda
}

is_possible_ideology_communist = {
	has_technology_researched = socialism
}

is_possible_ideology_anarchist = {
	has_technology_researched = anarchism
}

ideology_secular_humanitarian_valid_trigger = {
	OR = {
		is_interest_group_type = ig_intelligentsia
		is_interest_group_type = ig_trade_unions
	}
}

at_least_one_politician_is_no_fascist = {
	OR = {
		exists = ideology:ideology_fascist
		is_possible_ideology_fascist = yes
		any_scope_character = {
			has_role = politician
			NOT = { has_ideology = ideology:ideology_fascist }
		}
	}
}

at_least_one_politician_is_no_communist = {
	OR = {
		exists = ideology:ideology_communist
		is_possible_ideology_communist = yes
		any_scope_character = {
			has_role = politician
			NOT = { has_ideology = ideology:ideology_communist }
		}
	}
}

at_least_one_politician_is_no_anarchist = {
	OR = {
		exists = ideology:ideology_anarchist
		exists = ideology:ideology_anarcho_libertarian
		exists = ideology:ideology_libertarian_utopian
	}
	is_possible_ideology_anarchist = yes
	any_scope_character = {
		has_role = politician
		NAND = {
			has_ideology = ideology:ideology_anarchist
			has_ideology = ideology:ideology_anarcho_libertarian
			has_ideology = ideology:ideology_libertarian_utopian
		}
	}
}

country_is_a_judeo_christian_nation = {
	custom_tooltip = {
		text = country_is_a_judeo_christian_nation_tt
		religion = {
			OR = {
				has_discrimination_trait = christian
				has_discrimination_trait = jewish
			}
		}
	}
}

is_welfare_state_with_low_legitimacy = {
	has_law_or_variant = law_type:law_old_age_pension
	has_law_or_variant = law_type:law_worker_protections
	institution_investment_level = {
		institution = institution_social_security
		value = 5
	}
	institution_investment_level = {
		institution = institution_workplace_safety
		value = 5
	}
	government_legitimacy < 25
}
