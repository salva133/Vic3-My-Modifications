on_monthly_pulse_country = {
	on_actions = {
		on_monthly_pulse_country_mym_randoms
		#delay = { days = 4 }
		on_monthly_pulse_country_mym_events
		#delay = { days = 7 }
	}
}

albion = {
	effect = {
		if = {
			limit = {
				c:ABI ?= scope:target
			}
			scope:target = {
				set_up_albion_gov = yes
			}
		}
	}
}
