on_monthly_pulse_country_mym_events = {
	events = {
		anarchist_balkanisation.1
		no_assimilation.1
		australian_flavour.1
	}
	effect = {
		# Keep modifier_has_unhealthy_economy in sync with the economy flag.
		if = {
			limit = {
				has_healthy_economy = yes
				has_modifier = modifier_has_unhealthy_economy
			}
			remove_modifier = modifier_has_unhealthy_economy
		}
		else_if = {
			limit = {
				has_healthy_economy = no
				NOT = { has_modifier = modifier_has_unhealthy_economy }
			}
			add_modifier = modifier_has_unhealthy_economy
		}
	}
}

		on_monthly_pulse_country_mym_events

	}
}

# Apply the no-assimilation modifier to every country present at game start, so
# countries that begin multicultural have it from turn one instead of waiting for
# the first monthly pulse.
# Registered as a child on_action instead of attaching a competing effect = {} to the
# vanilla on_action: the on_actions list merges additively across files, whereas a second
# effect = {} would clash with vanilla's own (only one survives, "more than one effect").
on_game_started_after_lobby = {
	on_actions = { on_game_started_after_lobby_mym }
}

on_game_started_after_lobby_mym = {
	effect = {
		every_country = {
			mym_sync_no_assimilation_modifier = yes
		}
	}
}

# Make sure newly formed / released countries get the no-assimilation modifier
# synced to their citizenship law right away.
on_country_formed = {
	on_actions = { on_country_formed_mym }
}

on_country_formed_mym = {
	effect = {
		mym_sync_no_assimilation_modifier = yes
	}
}

# Sync the no-assimilation modifier the moment any law comes into force
# (enacted, imposed or switched to a variant), instead of waiting for the
# monthly pulse. Scope is the law; owner is the country.
on_law_activated = {
	on_actions = { on_law_activated_mym }
}

on_law_activated_mym = {
	effect = {
		owner = {
			mym_sync_no_assimilation_modifier = yes
		}
	}
}
