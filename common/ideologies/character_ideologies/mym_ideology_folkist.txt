ideology_folkist = {
    icon = gfx/interface/icons/ideology_icons/ideology_leader/ideology_folkist.dds

    character_ideology = yes

    lawgroup_citizenship = {
        law_subjecthood        = disapprove
        law_ethnostate         = strongly_approve
        law_national_supremacy = approve
        law_racial_segregation = neutral
        law_cultural_exclusion = neutral
        law_multicultural      = strongly_disapprove
    }
    lawgroup_labor_rights = {
        law_no_workers_rights  = strongly_disapprove
        law_regulatory_bodies  = neutral
        law_worker_protections = strongly_approve
    }
    lawgroup_welfare = {
        law_no_social_security = strongly_disapprove
        law_poor_laws       = disapprove
        law_wage_subsidies  = neutral
        law_old_age_pension = strongly_approve
    }
    lawgroup_church_and_state = {
        law_state_religion        = disapprove
        law_freedom_of_conscience = neutral
        law_total_separation      = approve
        law_state_atheism         = disapprove
    }
    lawgroup_childrens_rights = {
        law_child_labor_allowed       = disapprove
        law_restricted_child_labor    = neutral
        law_compulsory_primary_school = approve
    }
    lawgroup_rights_of_women = {
        law_no_womens_rights       = neutral
        law_women_own_property     = neutral
        law_women_in_the_workplace = disapprove
        law_womens_suffrage        = disapprove
    }

    country_trigger = {
        has_technology_researched = nationalism
    }

    interest_group_leader_trigger = {
        ideology_folkist_valid_trigger = yes
    }

    non_interest_group_leader_trigger = {
        # Agitator only spawns if at least one of the core folkist goals is not yet achieved
        NAND = {
            has_role = agitator
            owner = {
                has_law_or_variant = law_type:law_ethnostate
                has_law_or_variant = law_type:law_worker_protections
                has_law_or_variant = law_type:law_old_age_pension
            }
        }
        interest_group = {
            ideology_folkist_valid_trigger = yes
        }
    }

    interest_group_leader_weight = {
        value = 75

        # More likely if people are angry
        add = {
            value = owner.radical_fraction
            multiply = 100
        }

        # More likely with a convenient scapegoat (minority pops fuel ethnic resentment)
        if = {
            limit = {
                owner = {
                    any_scope_state = {
                        is_incorporated = yes
                        any_scope_pop = {
                            pop_acceptance < acceptance_status_4
                        }
                    }
                }
            }
            add = 25
            if = { # And if angry without ethnostate
                limit = {
                    owner = {
                        NOT = { has_law_or_variant = law_type:law_ethnostate }
                    }
                    ig_approval <= -5
                }
                subtract = {
                    value = ig_approval
                    multiply = 5
                }
            }
        }

        # Less likely in council republic
        if = {
            limit = {
                owner = {
                    has_law_or_variant = law_type:law_council_republic
                }
            }
            add = -75
        }

        # Less likely if content without ethnostate
        if = {
            limit = {
                owner = {
                    NOT = { has_law_or_variant = law_type:law_ethnostate }
                }
                ig_approval >= 5
            }
            subtract = {
                value = ig_approval
                multiply = 5
            }
        }

        # More likely for petty bourgeoisie and devout (Schönerer, Stoecker base)
        if = {
            limit = {
                OR = {
                    is_interest_group_type = ig_petty_bourgeoisie
                    is_interest_group_type = ig_devout
                }
            }
            add = 40
        }

        # Less likely for trade unions (prefer internationalist socialism)
        if = {
            limit = {
                is_interest_group_type = ig_trade_unions
            }
            add = -30
        }

        # Mass propaganda amplifies demagogic appeal
        if = {
            limit = {
                owner = {
                    has_technology_researched = mass_propaganda
                }
            }
            multiply = 1.5
        }

        # Suppressed if country committed to competing radical paths
        if = {
            limit = {
                OR = {
                    has_variable = communist_ig_var
                    has_variable = chose_vanguardism
                    has_variable = chose_anarchism
                    has_variable = chose_communism
                    has_variable = chose_fascism
                    has_variable = chose_ethno_nationalism
                }
            }
            multiply = 0.1
        }

        # Minority culture leaders won't champion ethnic exclusion against themselves
        if = {
            limit = {
                leader = {
                    culture = {
                        NOT = { is_primary_culture_of = prev.owner }
                    }
                }
            }
            multiply = 0
        }
    }

    non_interest_group_leader_weight = {
        value = 75

        # More likely if people are angry
        add = {
            value = owner.radical_fraction
            multiply = 100
        }

        # More likely with a convenient scapegoat
        if = {
            limit = {
                owner = {
                    any_scope_state = {
                        is_incorporated = yes
                        any_scope_pop = {
                            pop_acceptance < acceptance_status_4
                        }
                    }
                }
            }
            add = 25
            if = { # And if angry without ethnostate
                limit = {
                    owner = {
                        NOT = { has_law_or_variant = law_type:law_ethnostate }
                    }
                    interest_group.ig_approval <= -5
                }
                subtract = {
                    value = interest_group.ig_approval
                    multiply = 5
                }
            }
        }

        # Less likely in council republic
        if = {
            limit = {
                owner = {
                    has_law_or_variant = law_type:law_council_republic
                }
            }
            add = -75
        }

        # Less likely if content without ethnostate
        if = {
            limit = {
                owner = {
                    NOT = { has_law_or_variant = law_type:law_ethnostate }
                }
                interest_group.ig_approval >= 5
            }
            subtract = {
                value = interest_group.ig_approval
                multiply = 5
            }
        }

        # Mass propaganda amplifies demagogic appeal
        if = {
            limit = {
                owner = {
                    has_technology_researched = mass_propaganda
                }
            }
            multiply = 1.5
        }

        # Suppressed if country committed to competing radical paths
        if = {
            limit = {
                OR = {
                    has_variable = communist_ig_var
                    has_variable = chose_vanguardism
                    has_variable = chose_anarchism
                    has_variable = chose_communism
                    has_variable = chose_fascism
                    has_variable = chose_ethno_nationalism
                }
            }
            multiply = 0.1
        }
    }
}
