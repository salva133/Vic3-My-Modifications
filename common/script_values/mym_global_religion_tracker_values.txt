# Global population aggregated across every existing country
mym_global_population_sv = {
	value = 0
	every_country = {
		add = total_population
	}
}

# Per-heritage global population sums
# Each value is the sum, across every country, of total_population multiplied
# by the share of pops belonging to any religion that has the given heritage.

mym_heritage_christian_global_pop_sv = {
	value = 0
	every_country = {
		add = {
			value = total_population
			multiply = {
				value = 0
				add = "this.religion_percent_country(rel:catholic)"
				add = "this.religion_percent_country(rel:protestant)"
				add = "this.religion_percent_country(rel:orthodox)"
				add = "this.religion_percent_country(rel:oriental_orthodox)"
			}
		}
	}
}
