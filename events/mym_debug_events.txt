mym_debug.2 = { # Give all civilized countries era 4 tech & make them independent & Callback Ping
	type = country_event
	placement = ROOT
	title = mym_debug.2.t
	desc = mym_debug.2.d
	flavor = mym_debug.2.f

	is_popup = yes

	duration = 3

	event_image = {
		texture = gfx/event_images/bgy_debug.dds
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = gfx/interface/icons/event_icons/waving_flag.dds

	option = {
		name = mym_debug.2.a
		default_option = yes
		ai_chance = {
			base = 100
		}
	}
	option = {
		name = mym_debug.2.b
		ai_chance = {
			base = 0
		}

		trigger = {
			NOT = { has_variable = era_4_for_all_var }
		}

		custom_tooltip = {
			text = mym_debug.2.b_tt
			every_country = {
				limit = {
					NOT = { is_country_type = decentralized }
				}
				add_era_researched = era_4
				set_variable = era_4_for_all_var
			}
		}

		trigger_event = mym_debug.2
	}
	option = {
		name = mym_debug.2.c
		ai_chance = {
			base = 0
		}

		trigger = {
			any_country = {
				count > 0
				is_subject = yes
			}
		}

		custom_tooltip = {
			text = mym_debug.2.c_tt
			every_country = {
				limit = {
					NOT = { is_country_type = decentralized }
					is_subject = yes
				}
				make_independent = yes
			}
		}

		trigger_event = mym_debug.2
	}
	option = {
		name = mym_debug.2.e
		trigger = {
			ROOT = THIS
		}
		every_country = {
			limit = {
				NOT = { ROOT = THIS }
			}
			every_scope_state = {
				every_scope_pop = {
					limit = {
						culture = cu:spanish
						religion = rel:catholic
					}
					move_pop = ROOT.capital
				}
			}
		}

		trigger_event = mym_debug.2
	}
}
