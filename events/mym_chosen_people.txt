namespace = chosen_people

chosen_people.1 = {
	type = country_event
	hidden = yes

	trigger = {
		exists = c:PAL
		exists = c:ISR
		c:ISR = {
			NOT = { has_modifier = modifier_chosen_people }
			is_diplomatic_play_enemy_of = c:PAL
		}
	}

	immediate = {
		c:ISR = {
			add_modifier = modifier_chosen_people
		}
	}
}

chosen_people.2 = {
	type = country_event
	hidden = yes

	trigger = {
		NOT = { exists = c:PAL }
		exists = c:ISR
		c:ISR = {
			has_modifier = modifier_chosen_people
			NOT = { is_diplomatic_play_enemy_of = c:PAL }
		}
	}

	immediate = {
		c:ISR = {
			remove_modifier = modifier_chosen_people
		}
	}
}
