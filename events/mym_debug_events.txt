mym_debug.2 = {
	type = country_event
	placement = ROOT
	title = mym_debug.2.tt
	desc = mym_debug.2.dd
	flavor = mym_debug.2.ff



	event_image = { texture = gfx/event_pictures/mym_debug.dds }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = gfx/interface/icons/event_icons/waving_flag.dds

	immediate = {}

	# Closes the event
	option = {
		name = mym_debug.2.aa
		default_option = yes

		hidden_effect = {
			trigger_event = { id = mym_debug.7 popup = yes }
		}
	}

	# Gives Era 4 Tech to all civilized nations - Can be fired only once
	option = {
		name = mym_debug.2.a

		trigger = {
			NOT = { has_variable = era_5_for_all_var }
		}

		custom_tooltip = {
			text = mym_debug.2.b_tt
			every_country = {
				limit = {
					NOT = { is_country_type = decentralized }
				}
				add_era_researched = era_5
				set_variable = era_5_for_all_var
			}
		}

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}

	# Frees all civilized subject nations from their respective Overlord
	option = {
		name = mym_debug.2.b

		trigger = {
			any_country = {
				NOT = { is_country_type = decentralized }
				is_subject = yes
			}
		}

		custom_tooltip = {
			text = mym_debug.2.c_tt
			every_country = {
				limit = {
					NOT = { is_country_type = decentralized }
					is_subject = yes
				}
				make_independent = yes
			}
		}

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}

	option = { # Gives the root all the tech up to era 5 - clearly a cheat to circumvent achievement invalidation
		name = mym_debug.2.d

		trigger = { has_researchable_technology = yes }

		add_era_researched = era_5

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}
}

# MyM Debug Cheats Submenu - bundles the existing cheat-flavoured submenus
# (Leader Ideology Changer, Game Breaker, Character Spawn, Movement Attraction,
# Journal Entry Spawner) and exposes inline one-shot cheats. Reached from the
# main debug menu .4.
mym_debug.7 = {
	type = country_event
	placement = ROOT
	title = mym_debug.7.tt
	desc = mym_debug.7.dd
	flavor = mym_debug.7.ff

	event_image = { texture = gfx/event_pictures/mym_debug.dds }

	trigger = {} # Gets triggered by the main debug menu

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = gfx/interface/icons/event_icons/waving_flag.dds

	# Back to main menu
	option = {
		name = mym_debug.7.aa
		default_option = yes

		hidden_effect = {
			trigger_event = { id = mym_debug.4 popup = yes }
		}
	}

	# Leader Ideology Changer submenu
	option = {
		name = mym_debug.7.a

		trigger_event = { id = mym_debug.1 popup = yes }
	}

	# Game Breaker submenu
	option = {
		name = mym_debug.7.b

		trigger_event = { id = mym_debug.2 popup = yes }
	}

	# Character Spawn submenu
	option = {
		name = mym_debug.7.c

		trigger_event = { id = mym_debug.3 popup = yes }
	}

	# Movement Attraction Modifiers submenu
	option = {
		name = mym_debug.7.d

		trigger_event = { id = mym_debug.6 popup = yes }
	}

	# Journal Entry Spawner submenu
	option = {
		name = mym_debug.7.g

		trigger_event = { id = mym_debug.8 popup = yes }
	}

	# Inline cheat: Prussia swallows every European country whose primary
	# culture is north_german. Gated on ROOT being Prussia and at least one
	# valid target existing so the option is hidden when meaningless.
	option = {
		name = mym_debug.7.e

		trigger = {
			c:PRU ?= ROOT
			any_country = {
				NOT = { this = ROOT }
				country_has_primary_culture = cu:north_german
				country_is_in_europe = yes
			}
		}

		mym_cheat_pru_annex_north_german_europe = yes

		hidden_effect = {
			trigger_event = { id = mym_debug.7 popup = yes }
		}
	}

	# Inline cheat: scatter 100 levels each of rubber, oil and sulfur
	# buildings across ROOT's randomly chosen states. Capacity caps are
	# ignored on purpose - rebuilds after demolition are on the player.
	option = {
		name = mym_debug.7.f

		trigger = {
			any_scope_state = { owner = ROOT }
		}

		mym_cheat_spread_building_levels = { BUILDING = building_rubber_plantation }
		mym_cheat_spread_building_levels = { BUILDING = building_oil_rig }
		mym_cheat_spread_building_levels = { BUILDING = building_sulfur_mine }

		hidden_effect = {
			trigger_event = { id = mym_debug.7 popup = yes }
		}
	}
}
