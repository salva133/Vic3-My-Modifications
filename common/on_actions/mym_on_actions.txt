on_game_started = {
	on_actions = {
		on_game_started_mym_mennonites_setup
	}
}

on_monthly_pulse_country_mym_random_character_events = {
	random_events = {
		95 = hp_lovecraft.1
		5 = hp_lovecraft.2
	}
}

on_monthly_pulse_country_mym_character_events = {
	events = {
		stoker.1
		stoker.2

		hp_lovecraft.3

		milei.1

		nietzsche.1
		nietzsche.2
	}
}

on_monthly_pulse_country_mym_events = {
	events = {
		mennonite_migration.1

		anarchist_balkanisation.1
	}
	on_actions = {
		on_monthly_pulse_country_mym
	}
}

on_monthly_pulse_country_mym = {
	effect = {
		if = {
			limit = {
				has_healthy_economy = yes
				has_modifier = modifier_has_unhealthy_economy
			}
			remove_modifier = modifier_has_unhealthy_economy
		}
		else_if = {
			limit = {
				has_healthy_economy = no
				NOT = { has_modifier = modifier_has_unhealthy_economy }
			}
			add_modifier = modifier_has_unhealthy_economy
		}
	}
}

on_monthly_pulse_country_mym_heads_or_tails = {
	events = {
		heads_or_tails.1
	}
}

on_monthly_pulse_country = {
	on_actions = {
		delay = { days = 1 }
		on_monthly_pulse_country_mym_random_character_events
		delay = { days = 7 }
		on_monthly_pulse_country_mym_character_events
		delay = { days = 14 }
		on_monthly_pulse_country_mym_events
		delay = { days = 21 }
		on_monthly_pulse_country_mym_heads_or_tails
	}
}

on_game_started_mym_mennonites_setup = {
	effect = {
		# Eastern Europe
		on_game_started_mym_mennonites_setup_eastern_europe = yes

		# North America
		on_game_started_mym_mennonites_setup_north_america = yes
	}
}
