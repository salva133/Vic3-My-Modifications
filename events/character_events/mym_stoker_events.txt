namespace = stoker

stoker.1 = { # The Dmitry sets sails
	type = country_event
	placement = ROOT
	title =  stoker.1.tt
	desc =   stoker.1.dd
	flavor = stoker.1.ff

	event_image = {
		video = "votp_gunboat_diplomacy"
	}
	
	duration = 3
	
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	
	trigger = {
		game_date > 1885.7.24
		owns_entire_state_region = STATE_BULGARIA
		s:STATE_BULGARIA = {
			has_building = building_port
		}
		NOT = {
			has_global_variable = dmitry_set_sails_globvar
			has_global_variable = vlad_happened_globvar
		}
	}
	
	immediate = {
		set_global_variable = dmitry_set_sails_globvar
	}
	
	option = {
		name = stoker.1.a
		default_option = yes
	}
}

	flavor = stoker.2.ff

	}

	trigger = {
		game_date > 1885.8.6

		s:STATE_YORKSHIRE = {
			has_building = building_port
		}
		has_global_variable = dmitry_set_sails_globvar

	}
	
	option = {
		name = stoker.2.a
		default_option = yes

		create_character = {
			template = vlad_character_template
		}
		set_global_variable = vlad_happened_globvar
	}
}
