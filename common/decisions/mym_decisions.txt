want_to_be_free = {

	is_shown = {
		is_subject = yes
		is_player = yes
	}
	
	possible = {
		is_subject = yes
		is_at_war = no
		NOT = { is_diplomatic_play_enemy_of = overlord }
	}

	when_taken = {
		root = {
			create_diplomatic_play = {
				name = dp_independence
				type = dp_independence
				war = no
				target_country = overlord
				add_war_goal = {
					holder = ROOT
					type = war_reparations
					target_country = overlord
					primary_demand = no
				}
			}
		}
	}

	ai_chance = {
		base = 0
	}
}

release_all_subject = {

	is_shown = {
		is_player = yes
		is_subject = no
	}
	
	possible = {
		any_country = {
			is_subject = yes
		}
	}

	when_taken = {
		every_country = {
			limit = {
				NOT = { is_country_type = decentralized }
				is_subject = yes
			}
			make_independent = yes
		}
	}

	ai_chance = {
		base = 0
	}
}
