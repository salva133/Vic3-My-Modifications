# =====================================================================
# MyM Debug Menu
# ---------------------------------------------------------------------
# Entry point: decision_mym_debug_menu (common/decisions/mym_decisions.txt)
#
# Event ids follow the menu tree top-down:
#   .1 Main menu
#     .2 Cheats submenu
#       .3 Leader Ideology Changer
#       .4 Game Breaker
#       .5 Character Spawn
#       .6 Movement Attraction
#       .7 Journal Entry Spawner
#     .8 Misc Debug Functions
#
# Every submenu's default option is "Back" and re-opens its parent; the
# main menu's default option simply closes. Action options re-open their
# own submenu so several can be applied in a row.
# =====================================================================

# Main debug menu - reached from decision_mym_debug_menu.
mym_debug.1 = {
	type = country_event
	placement = ROOT
	title = mym_debug.1.tt
	desc = mym_debug.1.dd
	flavor = mym_debug.1.ff

	event_image = { texture = gfx/event_pictures/mym_debug.dds }

	trigger = {} # Gets triggered by decision

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = gfx/interface/icons/event_icons/waving_flag.dds

	# Closes the event with no additional effect
	option = {
		name = mym_debug.1.aa
		default_option = yes
	}
	# Cheats submenu (bundles the cheat-flavoured submenus and inline cheats)
	option = {
		name = mym_debug.1.a

		trigger_event = { id = mym_debug.2 popup = yes }
	}
	# Misc debug functions submenu
	option = {
		name = mym_debug.1.b

		trigger_event = { id = mym_debug.8 popup = yes }
	}
}

# Cheats submenu - bundles the cheat-flavoured submenus (Leader Ideology
# Changer, Game Breaker, Character Spawn, Movement Attraction, Journal Entry
# Spawner) and exposes inline one-shot cheats. Reached from the main menu .1.

	type = country_event
	placement = ROOT
	title = mym_debug.2.tt
	desc = mym_debug.2.dd
	flavor = mym_debug.2.ff

	event_image = { texture = gfx/event_pictures/mym_debug.dds }

	icon = gfx/interface/icons/event_icons/waving_flag.dds

	# Back to main menu

		name = mym_debug.2.aa
		default_option = yes

		hidden_effect = {
			trigger_event = { id = mym_debug.1 popup = yes }
		}
	}

	# Leader Ideology Changer submenu
	option = {
		name = mym_debug.2.a

		trigger_event = { id = mym_debug.3 popup = yes }

	option = {

	}

	# Journal Entry Spawner submenu
	option = {
		name = mym_debug.2.g

		trigger_event = { id = mym_debug.7 popup = yes }
	}

	# Inline cheat: Prussia swallows every European country whose primary
	# culture is north_german. Gated on ROOT being Prussia and at least one
	# valid target existing so the option is hidden when meaningless.
	option = {
		name = mym_debug.2.e

		trigger = {
			c:PRU ?= ROOT
			any_country = {
				NOT = { this = ROOT }
				country_has_primary_culture = cu:north_german
				country_is_in_europe = yes
			}
		}

		mym_cheat_pru_annex_north_german_europe = yes

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}

	# Inline cheat: scatter 100 levels each of rubber, oil and sulfur
	# buildings across ROOT's randomly chosen states. Capacity caps are
	# ignored on purpose - rebuilds after demolition are on the player.
	option = {
		name = mym_debug.2.f

		trigger = {
			any_scope_state = { owner = ROOT }
		}

		mym_cheat_spread_building_levels = { BUILDING = building_rubber_plantation }
		mym_cheat_spread_building_levels = { BUILDING = building_oil_rig }
		mym_cheat_spread_building_levels = { BUILDING = building_sulfur_mine }

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}
}

		default_option = yes

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}

	# Gives Era 5 Tech to all civilized nations - Can be fired only once
	option = {
		name = mym_debug.4.a

		trigger = {
			NOT = { has_variable = era_5_for_all_var }
		}

		custom_tooltip = {
			text = mym_debug.4.b_tt
			every_country = {
				limit = {
					NOT = { is_country_type = decentralized }
				}
				add_era_researched = era_5
				set_variable = era_5_for_all_var
			}
		}

		hidden_effect = {
			trigger_event = { id = mym_debug.4 popup = yes }
		}
	}

	# Frees all civilized subject nations from their respective Overlord
	option = {
		name = mym_debug.4.b

		trigger = {
			any_country = {
				NOT = { is_country_type = decentralized }
				is_subject = yes
			}
		}

		custom_tooltip = {
			text = mym_debug.4.c_tt
			every_country = {
				limit = {
					NOT = { is_country_type = decentralized }
					is_subject = yes
				}
				make_independent = yes
			}
		}

		hidden_effect = {
			trigger_event = { id = mym_debug.4 popup = yes }
		}
	}

	# Gives the root all the tech up to era 5 - clearly a cheat to circumvent achievement invalidation
	option = {
		name = mym_debug.4.d

		trigger = { has_researchable_technology = yes }

		add_era_researched = era_5

		hidden_effect = {
			trigger_event = { id = mym_debug.4 popup = yes }
		}
	}
}

# Journal Entry Spawner - force-adds a mod-owned journal entry to ROOT. Each
# option's trigger mirrors the journal entry's own possible block (plus a
# duplicate guard), so an entry can only be spawned when it could legitimately
# activate. Reached from the cheats submenu .2.
mym_debug.7 = {
	type = country_event
	placement = ROOT
	title = mym_debug.7.tt
	desc = mym_debug.7.dd
	flavor = mym_debug.7.ff

	event_image = { texture = gfx/event_pictures/mym_debug.dds }

	trigger = {} # Gets triggered by the cheats submenu

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = gfx/interface/icons/event_icons/waving_flag.dds

	# Back to cheats submenu
	option = {
		name = mym_debug.7.aa
		default_option = yes

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}

	# Adds the Gangs of New York journal entry
	option = {
		name = mym_debug.7.a

		trigger = {
			NOT = { has_journal_entry = je_gangs_of_new_york }
			# Mirrors je_gangs_of_new_york.possible
			game_date >= 1920.8.16
			s:STATE_NEW_YORK = { root_owns_state_in_region = yes }
			cultural_acceptance_base = {
				target = cu:south_italian
				value > acceptance_status_3
			}
		}

		add_journal_entry = { type = je_gangs_of_new_york }

		hidden_effect = {
			trigger_event = { id = mym_debug.7 popup = yes }
		}
	}

	# Adds the Baltic Dissolution journal entry
	option = {
		name = mym_debug.7.b

		trigger = {
			NOT = { has_journal_entry = je_mym_baltic_dissolution }
			# Mirrors je_mym_baltic_dissolution.possible
			c:UBD ?= this
			is_player = yes
			is_subject = no
			has_technology_researched = nationalism
			OR = {
				any_scope_state = { state_region = { is_homeland = cu:lithuanian } }
				any_scope_state = { state_region = { is_homeland = cu:latvian } }
				any_scope_state = { state_region = { is_homeland = cu:estonian } }
			}
		}

		add_journal_entry = { type = je_mym_baltic_dissolution }

		hidden_effect = {
			trigger_event = { id = mym_debug.7 popup = yes }
		}
	}

	# Adds the global religion tracker journal entry. je_mym_global_religion_tracker
	# has no possible block (it is a passive global tracker handed to every
	# country via on_actions), so there is nothing to mirror beyond the
	# duplicate guard.
	option = {
		name = mym_debug.7.c

		trigger = {
			NOT = { has_journal_entry = je_mym_global_religion_tracker }
		}

		add_journal_entry = { type = je_mym_global_religion_tracker }

		hidden_effect = {
			trigger_event = { id = mym_debug.7 popup = yes }
		}
	}
}
