mym_debug.2 = {
	type = country_event
	placement = ROOT
	title = mym_debug.2.tt
	desc = mym_debug.2.dd
	flavor = mym_debug.2.ff

	duration = 3

	event_image = {
		texture = gfx/event_images/bgy_debug.dds
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = gfx/interface/icons/event_icons/waving_flag.dds

	immediate = {
		random_primary_culture = {
			save_scope_as = our_primary_culture
		}
	}

	# Closes the event
	option = {
		name = mym_debug.2.a
		default_option = yes

		hidden_effect = {
			trigger_event = { id = mym_debug.4 popup = yes }
		}
	}

	# Gives Era 4 Tech to all civilized nations - Can be fired only once
	option = {
		name = mym_debug.2.b

		trigger = {
			NOT = { has_variable = era_4_for_all_var }
		}

		custom_tooltip = {
			text = mym_debug.2.b_tt
			every_country = {
				limit = {
					NOT = { is_country_type = decentralized }
				}
				add_era_researched = era_4
				set_variable = era_4_for_all_var
			}
		}

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}

	# Frees all civilized subject nations from their respective Overlord
	option = {
		name = mym_debug.2.c

		trigger = {
			any_country = {
				NOT = { is_country_type = decentralized }
				is_subject = yes
			}
		}

		custom_tooltip = {
			text = mym_debug.2.c_tt
			every_country = {
				limit = {
					NOT = { is_country_type = decentralized }
					is_subject = yes
				}
				make_independent = yes
			}
		}

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}

	# CALLBACK PING -- Pretty much that, calls all members of the primary culture
	# (a random one if there are multiple) and state religion 
	# to the capital state of the applying nation
	option = {
		name = mym_debug.2.e
		trigger = {
			ROOT = THIS
		}
		every_country = {
			limit = {
				NOT = { ROOT = THIS }
			}
			every_scope_state = {
				every_scope_pop = {
					limit = {
						culture = scope:our_primary_culture
						religion = root.religion
					}
					move_pop = ROOT.capital
				}
			}
		}

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}

	# Gives the root all the tech up to era 5 - clearly a cheat to circumvent achievement invalidation
	option = {
		name = mym_debug.2.g
		trigger = {
			has_researchable_technology = yes
		}
		add_era_researched = era_5

		hidden_effect = {
			trigger_event = { id = mym_debug.2 popup = yes }
		}
	}
}
