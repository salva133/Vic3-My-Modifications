ideology_archaeofuturist = {
	icon = gfx/interface/icons/ideology_icons/ideology_leader/ideology_folkist.dds

	character_ideology = yes

	lawgroup_citizenship = {
		law_ethnostate         = strongly_approve
		law_national_supremacy = approve
		law_racial_segregation = neutral
		law_cultural_exclusion = neutral
		law_multicultural      = strongly_disapprove
	}
	lawgroup_distribution_of_power = {
		law_anarchy            = strongly_disapprove
		law_universal_suffrage = strongly_disapprove
		law_census_voting      = disapprove
		law_wealth_voting      = neutral
		law_landed_voting      = neutral
		law_oligarchy          = approve
		law_autocracy          = approve
		law_technocracy        = approve
		law_single_party_state = neutral
	}
	lawgroup_governance_principles = {
		law_council_republic       = strongly_disapprove
		law_parliamentary_republic = disapprove
		law_presidential_republic  = disapprove
		law_theocracy              = disapprove
		law_social_monarchy        = neutral
		law_monarchy               = approve
		law_corporate_state        = strongly_approve
	}
	lawgroup_economic_system = {
		law_industry_banned      = strongly_disapprove
		law_cooperative_ownership = strongly_disapprove
		law_command_economy      = disapprove
		law_traditionalism       = disapprove
		law_agrarianism          = disapprove
		law_laissez_faire        = neutral
		law_interventionism      = approve
	}
	lawgroup_rights_of_women = {
		law_womens_suffrage        = disapprove
		law_women_in_the_workplace = disapprove
		law_women_own_property     = neutral
		law_no_womens_rights       = neutral
	}
	lawgroup_church_and_state = {
		law_state_religion        = disapprove
		law_state_atheism         = disapprove
		law_freedom_of_conscience = neutral
		law_total_separation      = approve
	}

	# Archaeofuturism enters history only once its doctrine has been articulated,
	# from 1998 onward
	country_trigger = {
		game_date >= 1998.1.1
	}

	interest_group_leader_trigger = {
		ideology_archaeofuturist_valid_trigger = yes
	}

	non_interest_group_leader_trigger = {
		# A non-leader archaeofuturist arises as an agitator with a grievance:
		# not where the ethnostate they demand is already law
		NAND = {
			has_role_of_type = agitator
			owner = {
				has_law_or_variant = law_type:law_ethnostate
			}
		}
		interest_group ?= {
			ideology_archaeofuturist_valid_trigger = yes
		}
	}

	interest_group_leader_weight = {
		value = 75

		# More likely if people are angry
		add = {
			value = owner.radical_fraction
			multiply = 100
		}

		# More likely with a convenient out-group to define against
		if = {
			limit = {
				owner = {
					any_scope_state = {
						is_incorporated = yes
						any_scope_pop = {
							pop_acceptance < acceptance_status_4
						}
					}
				}
			}
			add = 25
		}

		# Natural home among the intelligentsia and the officer caste
		if = {
			limit = {
				OR = {
					is_interest_group_type = ig_intelligentsia
					is_interest_group_type = ig_armed_forces
				}
			}
			add = 40
		}

		# Mass propaganda amplifies its metapolitical reach
		if = {
			limit = {
				owner = {
					has_technology_researched = mass_propaganda
				}
			}
			multiply = 1.5
		}

		# Suppressed if the country committed to a competing radical path
		if = {
			limit = {
				OR = {
					has_variable = communist_ig_var
					has_variable = chose_vanguardism
					has_variable = chose_anarchism
					has_variable = chose_communism
					has_variable = chose_fascism
				}
			}
			multiply = 0.1
		}

		# Minority-culture leaders won't champion an ethnostate against themselves
		if = {
			limit = {
				leader = {
					culture = {
						NOT = { is_primary_culture_of = prev.owner }
					}
				}
			}
			multiply = 0
		}
	}

	non_interest_group_leader_weight = {
		value = 75

		# More likely if people are angry
		add = {
			value = owner.radical_fraction
			multiply = 100
		}

		# More likely with a convenient out-group to define against
		if = {
			limit = {
				owner = {
					any_scope_state = {
						is_incorporated = yes
						any_scope_pop = {
							pop_acceptance < acceptance_status_4
						}
					}
				}
			}
			add = 25
		}

		# Mass propaganda amplifies its metapolitical reach
		if = {
			limit = {
				owner = {
					has_technology_researched = mass_propaganda
				}
			}
			multiply = 1.5
		}

		# Suppressed if the country committed to a competing radical path
		if = {
			limit = {
				OR = {
					has_variable = communist_ig_var
					has_variable = chose_vanguardism
					has_variable = chose_anarchism
					has_variable = chose_communism
					has_variable = chose_fascism
				}
			}
			multiply = 0.1
		}
	}
}
