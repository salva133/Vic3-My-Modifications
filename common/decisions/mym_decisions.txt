decision_mym_debug_menu = {

	is_shown = {
		is_player = yes
	}

	possible = {
		NOT = { has_variable = mym_debug_menu_open_var }
	}

	when_taken = {
		set_variable = mym_debug_menu_open_var
		trigger_event = { id = mym_debug.4 popup = yes }
	}

	ai_chance = {
		value = 0
	}
}

decision_want_to_be_free = {

	is_shown = {
		is_subject = yes
		is_player = yes
	}
	
	possible = {
		is_at_war = no
		NOT = { is_diplomatic_play_enemy_of = overlord }
	}

	when_taken = {
		root = {
			create_diplomatic_play = {
				name = dp_independence
				type = dp_independence
				war = no
				target_country = overlord
			}
		}
		change_infamy = 30
	}

	ai_chance = {
		value = 0
	}
}

decision_activate_callback_ping = {

	is_shown = {
		is_player = yes
		NOT = { has_modifier = modifier_callback_ping }
		total_population <= 10000000
	}

	possible = {
		NOT = { has_law = law_type:law_closed_borders }
		any_scope_state= {
			state_is_eligible_as_mass_migration_target = yes
		}
		any_country = {
			country_can_have_mass_migration_to = root
		}
	}

	when_taken = {
		add_modifier = modifier_callback_ping
	}
}

decision_deactivate_callback_ping = {

	is_shown = {
		is_player = yes
		has_modifier = modifier_callback_ping
		total_population <= 10000000
	}

	when_taken = {
		remove_modifier = modifier_callback_ping
	}
}
