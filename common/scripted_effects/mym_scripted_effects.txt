on_game_started_create_albionians = {
	add_homeland = cu:albionian
	every_scope_state = {
		convert_population = {
			target = rel:druidism
			value = 0.01
		}
		ordered_scope_pop = {
			change_pop_culture = {
				target = cu:albionian
				value = 1
			}
		}
	}
}

on_game_started_create_mennonites = {
	every_scope_state = {
		convert_population = {
			target = rel:mennonite
			value = 0.01
		}
	}
}

create_albion = {
	s:STATE_LANCASHIRE = {
		every_scope_state = {
			save_scope_as = lancashire_state
		}
	}
	if = {
		limit = { NOT = { exists = c:ABI } }
		create_country = {
			tag = ABI
			origin = c:GBR
			state = scope:lancashire_state
			on_created = {
				every_scope_state = {
					convert_population = {
						target = rel:druidism
						value = 0.9
					}
				}
			}
		}
	}
}

set_secret_police_ig_name_globally = {
	if = {
		limit = { exists = ig:ig_secret_police }
		set_secret_police_ig_name = yes
	}
}

set_secret_police_ig_name = {
	if = {
		limit = {
			country_has_primary_culture = cu:north_german
		}
		ig:ig_secret_police = {
			set_interest_group_name = ig_secret_police_north_german
		}
	}
	else_if = {
		limit = {
			country_has_primary_culture = cu:south_german
		}
		ig:ig_secret_police = {
			set_interest_group_name = ig_secret_police_south_german
		}
	}
	else_if = {
		limit = {
			country_has_primary_culture = cu:russian
			has_law = law_type:law_council_republic
		}
		ig:ig_secret_police = {
			set_interest_group_name = ig_secret_police_RUS_communist
		}
	}
	else_if = {
		limit = {
			country_has_primary_culture = cu:russian
		}
		ig:ig_secret_police = {
			set_interest_group_name = ig_secret_police_RUS
		}
	}
	else_if = {
		limit = {
			country_has_primary_culture = cu:romanian
		}
		ig:ig_secret_police = {
			set_interest_group_name = ig_secret_police_ROM
		}
	}
	else_if = {
		limit = {
			country_has_primary_culture = cu:finnish
		}
		ig:ig_secret_police = {
			set_interest_group_name = ig_secret_police_FIN
		}
	}
	else_if = {
		limit = {
			any_primary_culture = {
				has_discrimination_trait = heritage_iberian
			}
		}
		ig:ig_secret_police = {
			set_interest_group_name = ig_secret_police_SPA
		}
	}
	else_if = {
		limit = { c:ISR ?= this }
		ig:ig_secret_police = {
			set_interest_group_name = ig_secret_police_ISR
		}
	}
	else_if = {
		limit = {
			c:JAP ?= this
			c:JAP ?= { has_government_type = gov_fascist_state }
		}
		ig:ig_secret_police = {
			set_interest_group_name = ig_secret_police_ITA_fascist
		}
	}
	else_if = {
		limit = { c:ITA ?= this }
		ig:ig_secret_police = {
			set_interest_group_name = ig_secret_police_ITA
		}
	}
}
