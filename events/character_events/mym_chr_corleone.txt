namespace = corleone

corleone.1 = {
	type = country_event
	placement = root
	title = corleone.1.t
	desc = corleone.1.d
	flavor = corleone.1.f

	duration = 3

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		game_date > 1925.1.1
		exists = c:USA
		NOT = { has_variable = vito_andolini_var }
	}

	immediate = {
		create_character  = {
			template = vito_andolini_character_template
			on_created = {
				set_variable = vito_andolini_var
			}
		}
	}

	option = {
		# name = ""

	}
}
