namespace = economic_status

economic_status.1 = {
	type = country_event
	placement = ROOT
	title =  economic_status.1.t
	desc =   economic_status.1.d
	flavor = economic_status.1.f

	event_image = {
		video = "southamerica_public_figure_assassination"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = gfx/interface/icons/event_icons/waving_flag.dds

	duration = 1

	trigger = {
		has_law = law_type:law_economic_status_2
		standard_of_living < 5
	}

	option = {
		name = economic_status.1.a
		default_option = yes

		activate_law = law_type:law_economic_status_1
	}
}
